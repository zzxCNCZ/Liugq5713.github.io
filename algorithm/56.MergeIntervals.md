# 56. Merge Intervals

```js
var merge = function(intervals) {
  if (intervals.length == 0) {
    return intervals;
  }
  // 基于 数组第一个元素 排序
  intervals.sort((a, b) => a[0] - b[0]);
  let res = [];
  res.push(
    intervals.reduce((acc, cur) => {
      // 数组交叉或者包含
      if (acc[1] >= cur[0]) {
        // 交叉的情况
        if (acc[1] < cur[1]) {
          acc[1] = cur[1];
        }
        // 包含则返回目前的acc,即跳过这个数组
        return acc;
      } else {
        // 完全不关联，更新 acc
        res.push(acc);
        return cur;
      }
    })
  );
  return res;
};
```
